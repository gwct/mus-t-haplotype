16a.Copy sequences that pass the filters so snakemake doesn't error out

    --- New tree run, no pahari  ---

    WD:     $PROJECT_ROOT/analysis/02-mus-t-windows-new-tree/03-fasta-no-anc-filter-no-pahari/

    IN:     $PROJECT_ROOT/data/mm10-10kb-window-stats-new-tree-no-pahari.tsv
    IN:     $PROJECT_ROOT/analysis/02-mus-t-windows-new-tree/03-fasta-no-anc-filter-no-pahari/chr17-filter/

    FILTERS:    - All sequences must be present in the alignment after seq filtering (5); column 33 of the alignment summary tsv file
                - At least 4 of the sequences in the alignment must be unique (so IQtree can infer the tree); column 34 of the alignment summary tsv file
                - There must be 0 sequences comprised of all gap or missing characters (else IQtree errors out); column 43 of the alignment summary tsv file

    CMD:    grep -v "#" ../../../data/mm10-10kb-window-stats-new-tree-no-pahari.tsv | grep -v start | awk '$33==5 && $34>=4 && $43 == 0 {print $1}' > used-windows.txt
    CMD:    while read line; do cp chr17-filter/$line.filter.fa chr17-filter-used/$line.filter.fa; done < used-windows.txt

    OUT:    $PROJECT_ROOT/analysis/02-mus-t-windows-new-tree/03-fasta-no-anc-filter-no-pahar/used-windows.txt
    OUT:    $PROJECT_ROOT/analysis/02-mus-t-windows-new-tree/03-fasta-no-anc-filter-no-pahari/chr17-filter-used/
